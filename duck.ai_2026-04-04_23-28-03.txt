import tkinter as tk
from tkinter import scrolledtext

class ChatUI(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Chat Interface")
        self.geometry("900x600")
        self.minsize(600, 400)

        # Main paned layout: left = chat area, right = optional logs
        self.paned = tk.PanedWindow(self, orient=tk.HORIZONTAL, sashrelief=tk.RAISED)
        self.paned.pack(fill=tk.BOTH, expand=True)

        # Chat frame (left)
        self.chat_frame = tk.Frame(self.paned)
        self.paned.add(self.chat_frame, minsize=300)

        # Top toolbar with toggle button for logs
        toolbar = tk.Frame(self.chat_frame)
        toolbar.pack(fill=tk.X, side=tk.TOP)
        self.toggle_btn = tk.Button(toolbar, text="Toggle Logs", command=self.toggle_logs)
        self.toggle_btn.pack(side=tk.RIGHT, padx=4, pady=4)

        # Chat history (large)
        history_frame = tk.Frame(self.chat_frame)
        history_frame.pack(fill=tk.BOTH, expand=True, padx=6, pady=(0,6))

        self.chat_history = scrolledtext.ScrolledText(history_frame, wrap=tk.WORD, state=tk.DISABLED)
        self.chat_history.pack(fill=tk.BOTH, expand=True)

        # Bottom input area
        input_frame = tk.Frame(self.chat_frame)
        input_frame.pack(fill=tk.X, side=tk.BOTTOM, padx=6, pady=6)

        # Expandable Text widget with send button
        self.input_text = tk.Text(input_frame, height=1, wrap=tk.WORD)
        self.input_text.pack(side=tk.LEFT, fill=tk.X, expand=True)
        self.input_text.bind("<KeyRelease>", self._auto_expand)
        self.input_text.bind("<Control-Return>", self._send_from_ctrl_enter)

        send_btn = tk.Button(input_frame, text="Send", width=10, command=self._send_message)
        send_btn.pack(side=tk.RIGHT, padx=(6,0))

        # Right-side log panel (initially hidden)
        self.log_frame = tk.Frame(self.paned, width=250)
        self.log_visible = False
        # Log content
        self.logs = scrolledtext.ScrolledText(self.log_frame, wrap=tk.WORD, state=tk.DISABLED, width=30)
        self.logs.pack(fill=tk.BOTH, expand=True, padx=4, pady=4)

        # Auto-expand limits
        self.min_input_lines = 1
        self.max_input_lines = 8

        # Example: add a few messages
        self._append_history("System: Chat initialized.")
        self._append_log("Log: UI started.")

    def toggle_logs(self):
        if self.log_visible:
            try:
                self.paned.forget(self.log_frame)
            except Exception:
                pass
            self.log_visible = False
        else:
            # add log frame to the paned window on the right
            self.paned.add(self.log_frame, minsize=150)
            self.log_visible = True

    def _append_history(self, text):
        self.chat_history.configure(state=tk.NORMAL)
        self.chat_history.insert(tk.END, text + "\n")
        self.chat_history.see(tk.END)
        self.chat_history.configure(state=tk.DISABLED)

    def _append_log(self, text):
        self.logs.configure(state=tk.NORMAL)
        self.logs.insert(tk.END, text + "\n")
        self.logs.see(tk.END)
        self.logs.configure(state=tk.DISABLED)

    def _send_message(self):
        msg = self.input_text.get("1.0", tk.END).rstrip("\n")
        if not msg.strip():
            return
        self._append_history("You: " + msg)
        self._append_log("Sent message of length %d" % len(msg))
        # clear input and reset size
        self.input_text.delete("1.0", tk.END)
        self._set_input_lines(self.min_input_lines)

    def _send_from_ctrl_enter(self, event=None):
        # allow ctrl+enter to send (useful for multiline)
        self._send_message()
        return "break"

    def _auto_expand(self, event=None):
        # Adjust Text widget height by counting display lines
        content = self.input_text.get("1.0", tk.END).rstrip("\n")
        if content == "":
            lines = self.min_input_lines
        else:
            # count visual lines by splitting on newline and measuring wrap
            font = self.input_text.cget("font")
            # approximate: use number of '\n' plus wrapped lines according to widget width
            raw_lines = content.split("\n")
            try:
                widget_width = int(self.input_text.winfo_width() / self.input_text.winfo_fpixels('1c'))
            except Exception:
                widget_width = 80
            lines = 0
            for rl in raw_lines:
                if rl == "":
                    lines += 1
                else:
                    # approximate wrap: characters per line equals widget width in chars
                    chars_per_line = max(widget_width, 20)
                    wrapped = (len(rl) - 1) // chars_per_line + 1
                    lines += wrapped
            lines = max(self.min_input_lines, lines)
        lines = min(lines, self.max_input_lines)
        self._set_input_lines(lines)

    def _set_input_lines(self, n):
        self.input_text.configure(height=n)

if __name__ == "__main__":
    app = ChatUI()
    app.mainloop()

        Why do we insert as a single paragraph (ending with newline)? Because Text widget
        applies 'justify' per-paragraph; ensuring a trailing newline causes the whole message
        to be treated as one paragraph.

        Set Text widget height in lines. We track the last set value to avoid redundant
        layout updates (which can cause flicker).

        self.input_text.configure(height=n)
        self._last_line_count = n

    def _auto_expand(self, event=None):
        """
        Compute required number of visual lines for the current input content and resize the
        input Text widget accordingly, clamped between min_input_lines and max_input_lines.

        What we measure and why:
        - The Text widget supports word-wrapping, so a long logical line can span multiple visual
          lines. To estimate visual lines we:
          1) Measure the widget pixel width.
          2) Measure average character pixel width for the current font.
          3) Estimate characters-per-line = widget_width_px / avg_char_width.
          4) For each logical line (split by '\n') compute how many wrapped lines it produces.
        - This is an approximation but is far better than purely counting '\n' because it accounts
          for wrapping.
        - We update only when the computed line count changes to avoid excessive layout churn.
        """
        # Get content without the trailing newline Tk inserts
        content = self.input_text.get("1.0", "end-1c")
        if content == "":
            lines = self.min_input_lines
        else:
            # Determine widget pixel width. If it's not realized yet (very small),
            # fall back to an approximate width computed from the widget 'width' config.
            widget_width_px = self.input_text.winfo_width()
            if widget_width_px <= 1:
                try:
                    widget_width_chars = int(self.input_text.cget("width"))
                except Exception:
                    widget_width_chars = 80
                avg_char_width = self.input_font.measure("n") or 7
                widget_width_px = widget_width_chars * avg_char_width

            # Average character width in pixels for the chosen font.
            avg_char_width = max(1, self.input_font.measure("n"))

            # Approximate number of characters that fit in one visual line.
            chars_per_line = max(20, int(widget_width_px / avg_char_width))

            # Count wrapped visual lines by examining each logical line.
            raw_lines = content.split("\n")
            lines = 0
            for rl in raw_lines:
                if rl == "":
                    # Blank logical line still consumes one visual line
                    lines += 1
                else:
                    # integer math to compute wrapped lines: ceil(len / chars_per_line)
                    wrapped = (len(rl) - 1) // chars_per_line + 1
                    lines += wrapped
            lines = max(self.min_input_lines, lines)

        # Clamp to maximum allowed visible lines
        lines = min(lines, self.max_input_lines)

        # Only update the widget if the line count changed to avoid layout thrash.
        if lines != self._last_line_count:
            self._set_input_lines(lines)
            # Ensure the layout and geometry managers update immediately to avoid flicker.
            self.input_text.update_idletasks()
